const YES = 1;
const NO = 0;

procedure set_a_and_b(a_, b_);
procedure do_the_hard_computations();
function gimme_a();
function gimme_b();
function are_a_and_b_coprime();
function gimme_nontrivial_divisor();
function gimme_a_multiplier();
function gimme_b_multiplier();

main {
    read a, b;
    call set_a_and_b(a, b);
    call do_the_hard_computations();
    call ans = are_a_and_b_coprime();
    write ans;
    if ans {
       call a_multiplier = gimme_a_multiplier();
       call b_multiplier = gimme_b_multiplier();
       write a_multiplier, b_multiplier;
    } else {
       call d = gimme_nontrivial_divisor();
       write d;
    }
}
